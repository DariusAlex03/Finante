Schema bazei de date Supabase

Tabelele de care ai nevoie ca fiecare utilizator să-și vadă doar propriile date — cu SQL-ul gata de copiat în Supabase.

← Înapoi la autentificare

01Pașii de configurare

De la zero la funcțional în ~5 minute.

Creează proiectul

Intră pe supabase.com → New project. Alege o parolă pentru baza de date și regiunea eu-central (Frankfurt, cea mai aproape).

Rulează SQL-ul

În meniul din stânga: SQL Editor → New query → lipește scriptul complet de la secțiunea 03 → Run. Creează toate tabelele + regulile de securitate dintr-un foc.

Activează autentificarea pe email

Authentication → Sign In / Up — Email e activat implicit. Pentru testare rapidă poți opri "Confirm email" (utilizatorii intră fără să confirme mailul).

Copiază cheile

Settings → API → copiază Project URL și cheia anon public.

Pune-le în aplicație

Deschide fișierul supabase-config.js și completează cele două valori. Gata — pagina de autentificare va folosi conturi reale.

03Scriptul SQL complet

Copiază tot și rulează-l o singură dată în SQL Editor. Include tabelele, indexurile și securitatea pe rânduri (RLS).

De ce contează RLS: fără Row Level Security, orice utilizator autentificat ar putea citi datele tuturor. Politicile de mai jos garantează că fiecare om își vede doar propriile rânduri.
-- ============================================
-- BaniBine · Schema completă pentru Supabase
-- Rulează o singură dată în SQL Editor
-- ============================================

-- 1) Tranzacții (venituri + cheltuieli)
create table transactions (
  id          uuid primary key default gen_random_uuid(),
  user_id     uuid not null references auth.users(id) on delete cascade,
  type        text not null check (type in ('income', 'expense')),
  amount      numeric(12,2) not null check (amount > 0),
  category    text not null,
  description text,
  date        date not null default current_date,
  created_at  timestamptz not null default now()
);

-- 2) Obiective de economisire
create table goals (
  id         uuid primary key default gen_random_uuid(),
  user_id    uuid not null references auth.users(id) on delete cascade,
  name       text not null,
  target     numeric(12,2) not null check (target > 0),
  saved      numeric(12,2) not null default 0,
  color      text default '#d97757',
  deadline   date,
  created_at timestamptz not null default now()
);

-- 3) Bugete lunare pe categorie
create table budgets (
  id       uuid primary key default gen_random_uuid(),
  user_id  uuid not null references auth.users(id) on delete cascade,
  category text not null,
  amount   numeric(12,2) not null check (amount >= 0),
  unique (user_id, category)
);

-- 4) Împrumuturi date
create table loans (
  id         uuid primary key default gen_random_uuid(),
  user_id    uuid not null references auth.users(id) on delete cascade,
  person     text not null,
  amount     numeric(12,2) not null check (amount > 0),
  note       text,
  date       date not null default current_date,
  created_at timestamptz not null default now()
);

-- 5) Plăți primite pentru împrumuturi
create table loan_payments (
  id      uuid primary key default gen_random_uuid(),
  loan_id uuid not null references loans(id) on delete cascade,
  user_id uuid not null references auth.users(id) on delete cascade,
  amount  numeric(12,2) not null check (amount > 0),
  date    date not null default current_date
);

-- 6) Setări per utilizator
create table settings (
  user_id          uuid primary key references auth.users(id) on delete cascade,
  starting_balance numeric(12,2) not null default 0
);

-- Indexuri pentru interogări rapide
create index idx_tx_user_date on transactions (user_id, date desc);
create index idx_goals_user on goals (user_id);
create index idx_budgets_user on budgets (user_id);
create index idx_loans_user on loans (user_id);
create index idx_loan_payments_loan on loan_payments (loan_id);

-- ============================================
-- Row Level Security: fiecare își vede DOAR datele lui
-- ============================================
alter table transactions  enable row level security;
alter table goals         enable row level security;
alter table budgets       enable row level security;
alter table loans         enable row level security;
alter table loan_payments enable row level security;
alter table settings      enable row level security;

create policy "datele mele" on transactions
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);

create policy "datele mele" on goals
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);

create policy "datele mele" on budgets
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);

create policy "datele mele" on loans
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);

create policy "datele mele" on loan_payments
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);

create policy "datele mele" on settings
  for all using (auth.uid() = user_id) with check (auth.uid() = user_id);
